function saturation(pixels, adj) {
  let d = pixels.data;
  adj = adj < -1 ? -1 : adj;
  for (let i = 0; i < d.length; i += 4) {
    const r = d[i];
    const g = d[i + 1];
    const b = d[i + 2];
    // weights from CCIR 601 spec
    const gray = 0.2989 * r + 0.587 * g + 0.114 * b;
    d[i] = -gray * adj + d[i] * (1 + adj);
    d[i + 1] = -gray * adj + d[i + 1] * (1 + adj);
    d[i + 2] = -gray * adj + d[i + 2] * (1 + adj);
  }
}
